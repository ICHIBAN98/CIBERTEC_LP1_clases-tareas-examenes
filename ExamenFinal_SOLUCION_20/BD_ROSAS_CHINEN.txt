drop database BD_ROSAS_CHINEN;
create database BD_ROSAS_CHINEN;
use BD_ROSAS_CHINEN;

CREATE TABLE proveedor (
	id int NOT NULL AUTO_INCREMENT,
    ruc varchar(100),
    nombre varchar(100),
	PRIMARY KEY (id)
);

CREATE TABLE libro (
	id int NOT NULL AUTO_INCREMENT,
    codigo varchar(100),
    nombreLibro varchar(100),
    stock int,
	PRIMARY KEY (id)
);

CREATE TABLE venta (
	id int NOT NULL AUTO_INCREMENT,
    fechaRegistro datetime,
    idProveedor int,
    idLibro int,
	PRIMARY KEY (id)
);
